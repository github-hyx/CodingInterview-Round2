class Solution {
public:
    bool Find(int target, vector<vector<int> > array) 
    {
        if(array.size()==0)
            return false;
        int rows=array.size();
        int cols=array[0].size();
        int row=0;
        int col=cols-1;//注意行数和列数是从0开始计算的！！
        while(row<=rows-1&&col>=0)
        {
            if(array[row][col]<target)
                row++;
            else if(array[row][col]>target)
                col--;
            else
                return true;
        }
        return false;
    }
};
